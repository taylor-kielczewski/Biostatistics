  mutate(
    sink_c = sink_rate - mean(sink_rate, na.rm = TRUE),
    microbe_c = microbial_abundance - mean(microbial_abundance, na.rm = TRUE),
    detriti_c = detritivore_biomass - mean(detritivore_biomass, na.rm = TRUE)
